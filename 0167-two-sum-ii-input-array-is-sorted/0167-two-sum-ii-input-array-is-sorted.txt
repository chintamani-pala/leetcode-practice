func twoSum(numbers []int, target int) []int {
    low := 0
    high := len(numbers) - 1
    for {
        if low>=high{
            break
        }
        total:=numbers[low] + numbers[high]
        if total == target {
            return []int{low+1, high+1}
        }
        if total < target{
            low += 1
        }
        if total > target {
            high -= 1
        }
    }
    return []int{0,0}
}
