func findMaxAverage(nums []int, k int) float64 {
    sum := 0

    //first k sum
    for i:=0;i<k;i++ {
        sum += nums[i]
    }
    maximum := sum
    
    for i:=k;i<len(nums);i++ {
        sum += nums[i] - nums[i-k]
        maximum = max(maximum, sum)
    }

    return float64(maximum)/float64(k)
}
