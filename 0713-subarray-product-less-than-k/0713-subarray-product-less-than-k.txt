func numSubarrayProductLessThanK(nums []int, k int) int {
    if k<=1{
        return 0
    }
    multi := 1
    count := 0
    l:=0
    length := len(nums)
    for r:=0; r<length;r++{
        multi = multi*nums[r]
        for multi >= k && l<length{
            multi = multi / nums[l]
            l++
        }
        count += r-l + 1
    }
    return count
}
